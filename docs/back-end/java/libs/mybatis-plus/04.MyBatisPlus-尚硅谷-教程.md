## 1. 快速开始

### 1.1. 数据库

```sql
create database mybatis_plus_demo;

use mybatis_plus_demo;

create table `user` (
  `id` bigint not null,
  `name` varchar(30),
  `age` int,
  `email` varchar(50),
  primary key (`id`)
) engine=InnoDB;

insert into user(id, name, age, email) values
(1, 'Jone', 18, 'test1@qq.comm'),
(2, 'Jack', 20, 'test2@qq.comm'),
(3, 'Tom', 28, 'test3@qq.comm'),
(4, 'Sandy', 21, 'test4@qq.comm'),
(5, 'Billie', 24, 'test5@qq.comm');
```

### 1.4. mapper

实体类:

```java
@Data
public class User {
    private Long id;
    private String name;
    private Integer age;
    private String email;
}
```

mapper 接口:

```java
public interface UserMapper extends BaseMapper<User> {
}
```

配置:

```java
@SpringBootApplication
// 扫描指定包下的 mapper
@MapperScan("org.example.mybatisplusdemo.mapper")
public class MyBatisPlusDemoApplication {
    public static void main(String[] args) {
        SpringApplication.run(MyBatisPlusDemoApplication.class, args);
    }
}
```

测试:

```java
@SpringBootTest
class MyBatisPlusDemoApplicationTests {
    @Autowired
    private UserMapper userMapper;

    @Test
    void testSelectList() {
        // 传入 null 则不设置条件，即查询所有
        List<User> list = userMapper.selectList(null);
        list.forEach(System.out::println);
    }
}
```

## 2. BaseMapper

```java
public interface BaseMapper<T> extends Mapper<T> {
    int insert(T entity);

    int deleteById(Serializable id);
    int deleteById(T entity);
    int deleteByMap(@Param(Constants.COLUMN_MAP) Map<String, Object> columnMap);
    int delete(@Param(Constants.WRAPPER) Wrapper<T> queryWrapper);
    int deleteBatchIds(@Param(Constants.COLL) Collection<?> idList);

    int updateById(@Param(Constants.ENTITY) T entity);
    int update(@Param(Constants.ENTITY) T entity, @Param(Constants.WRAPPER) Wrapper<T> updateWrapper);
    
    T selectById(Serializable id);
    List<T> selectBatchIds(@Param(Constants.COLL) Collection<? extends Serializable> idList);
    List<T> selectByMap(@Param(Constants.COLUMN_MAP) Map<String, Object> columnMap);
    T selectOne(@Param(Constants.WRAPPER) Wrapper<T> queryWrapper)
    boolean exists(Wrapper<T> queryWrapper);
    Long selectCount(@Param(Constants.WRAPPER) Wrapper<T> queryWrapper);
    List<T> selectList(@Param(Constants.WRAPPER) Wrapper<T> queryWrapper);
    List<Map<String, Object>> selectMaps(@Param(Constants.WRAPPER) Wrapper<T> queryWrapper);
    List<Object> selectObjs(@Param(Constants.WRAPPER) Wrapper<T> queryWrapper);
    <P extends IPage<T>> P selectPage(P page, @Param(Constants.WRAPPER) Wrapper<T> queryWrapper);
    <P extends IPage<Map<String, Object>>> P selectMapsPage(P page, @Param(Constants.WRAPPER) Wrapper<T> queryWrapper);
}
```

### 2.1. 增

```java
@Test
void testInsert() {
    User user = new User();
    user.setName("张三");
    user.setAge(18);
    user.setEmail("zhangsan@qq.com");

    // INSERT INTO user ( id, name, age, email ) VALUES ( ?, ?, ?, ? )
    int result = userMapper.insert(user);

    System.out.println("result:" + result);

    // 插入后可以获取到 id
    System.out.println("id:" + user.getId());
}
```

### 2.2. 删

```java
@Test
void testDeleteById() {
    // DELETE FROM user WHERE id=?
    int result = userMapper.deleteById(2047969882462093314L);
    System.out.println("result:" + result);
}

// 根据 map 中设置的条件来删除:  key 为 列名，value 为 列的值
@Test
void testDeleteByMap() {
    Map<String, Object> map = new HashMap<>();
    map.put("name", "张三");
    map.put("age", 18);

    // DELETE FROM user WHERE name = ? AND age = ?
    int result = userMapper.deleteByMap(map);
    System.out.println("result:" + result);
}

// 根据 id 集合来批量删除
@Test
void testDeleteBatchIds() {
    // DELETE FROM user WHERE id IN ( ? , ? , ? )
    int result = userMapper.deleteBatchIds(Arrays.asList(11, 22, 33));
    System.out.println("result:" + result);
}
```

### 2.3. 改

```java
// 根据 id 来定位并更新记录，仅更新 非 null 的字段
@Test
void testUpdateById() {

}
```
