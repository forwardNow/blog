# MouseEvent

```typescript
interface MouseEvent extends UIEvent {
  readonly altKey: boolean;
  readonly ctrlKey: boolean;
  readonly metaKey: boolean;
  readonly shiftKey: boolean;

  readonly clientX: number; // The X coordinate of the mouse pointer in local (DOM content) coordinates.
  readonly clientY: number;

  readonly movementX: number; // The X coordinate of the mouse pointer relative to the position of the last mousemove event.
  readonly movementY: number;

  readonly offsetX: number;
  readonly offsetY: number;

  readonly pageX: number; // The X coordinate of the mouse pointer relative to the whole document.
  readonly pageY: number;

  readonly x: number; // Alias for MouseEvent.clientX
  readonly y: number;
}
```
